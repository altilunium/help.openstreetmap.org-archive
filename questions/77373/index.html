Edit:

https://overpass-turbo.eu/s/ZH6

[bbox:{{bbox}}];
(
  nw[building=school];
  -
  (
    way[amenity=school];map_to_area;
    nw(area)[building=school];
  );
)->.Outside;  
foreach .Outside (
  way.Outside(around:100);
  way._(if:count(ways) == 1);
  out center;
);
