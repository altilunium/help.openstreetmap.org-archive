Try this:

[out:json][timeout:25];   
{{geocodeArea:Zurich}}->.searchArea;    
(    
  node[uic_ref=8503011]["railway"="station"];  
  node[uic_ref=8503000]["railway"="station"];  
  rel(bn)(area.searchArea);  
  way(r)(area.searchArea);  
);      
out body;    
>;      
out skel qt;
